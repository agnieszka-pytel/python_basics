In [187]:
t = np.linspace (0, 2*np.pi, 100, endpoint=False)
x = 16*np.sin (t)**3
y = 13*np.cos(t) - 5*np.cos(2*t) - 2*np.cos(3*t) - np.cos (4*t)
h = ['*','/','|','-','x','o','0','.']
for i in range(0,len(h)):
    plt.subplot(3,3,i+1)
    plt.fill (x, y, fill = False, hatch=h[i])
    plt.axis("equal")
plt.subplots_adjust(wspace = 0.5, hspace = 0.5)
plt.show()

plt.subplot(121)
n = tips['smoker'].value_counts()['Yes']
rozklad = [n, len(tips)-n]
etykiety = ['Yes','No']
x = np.arange(len(rozklad))
